299void Raytracing::createSBT() {
300 // load in functionality for raytracing shader group handles
301 PFN_vkGetRayTracingShaderGroupHandlesKHR
302 pvkGetRayTracingShaderGroupHandlesKHR =
303 (PFN_vkGetRayTracingShaderGroupHandlesKHR)vkGetDeviceProcAddr(
304 device->getLogicalDevice(),
305 "vkGetRayTracingShaderGroupHandlesKHR");
306
307 raytracing_properties = VkPhysicalDeviceRayTracingPipelinePropertiesKHR{};
308 raytracing_properties.sType =
309 VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_RAY_TRACING_PIPELINE_PROPERTIES_KHR;
310
311 VkPhysicalDeviceProperties2 properties{};
312 properties.sType = VK_STRUCTURE_TYPE_PHYSICAL_DEVICE_PROPERTIES_2;
313 properties.pNext = &raytracing_properties;
314
315 vkGetPhysicalDeviceProperties2(device->getPhysicalDevice(), &properties);
316
317 uint32_t handle_size = raytracing_properties.shaderGroupHandleSize;
318 uint32_t handle_size_aligned =
319 align_up(handle_size, raytracing_properties.shaderGroupHandleAlignment);
320
321 uint32_t group_count = static_cast<uint32_t>(shader_groups.size());
322 uint32_t sbt_size = group_count * handle_size_aligned;
323
324 std::vector<uint8_t> handles(sbt_size);
325
326 VkResult result = pvkGetRayTracingShaderGroupHandlesKHR(
327 device->getLogicalDevice(), graphicsPipeline, 0, group_count, sbt_size,
328 handles.data());
329 ASSERT_VULKAN(result, "Failed to get ray tracing shader group handles!")
330
331 const VkBufferUsageFlags bufferUsageFlags =
332 VK_BUFFER_USAGE_SHADER_BINDING_TABLE_BIT_KHR |
333 VK_BUFFER_USAGE_SHADER_DEVICE_ADDRESS_BIT;
334 const VkMemoryPropertyFlags memoryUsageFlags =
335 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT |
336 VK_MEMORY_PROPERTY_HOST_COHERENT_BIT;
337
338 raygenShaderBindingTableBuffer.create(device, handle_size, bufferUsageFlags,
339 memoryUsageFlags);
340
341 missShaderBindingTableBuffer.create(device, 2 * handle_size, bufferUsageFlags,
342 memoryUsageFlags);
343
344 hitShaderBindingTableBuffer.create(device, handle_size, bufferUsageFlags,
345 memoryUsageFlags);
346
347 void* mapped_raygen = nullptr;
348 vkMapMemory(device->getLogicalDevice(),
349 raygenShaderBindingTableBuffer.getBufferMemory(), 0,
350 VK_WHOLE_SIZE, 0, &mapped_raygen);
351
352 void* mapped_miss = nullptr;
353 vkMapMemory(device->getLogicalDevice(),
354 missShaderBindingTableBuffer.getBufferMemory(), 0, VK_WHOLE_SIZE,
355 0, &mapped_miss);
356
357 void* mapped_rchit = nullptr;
358 vkMapMemory(device->getLogicalDevice(),
359 hitShaderBindingTableBuffer.getBufferMemory(), 0, VK_WHOLE_SIZE,
360 0, &mapped_rchit);
361
362 memcpy(mapped_raygen, handles.data(), handle_size);
363 memcpy(mapped_miss, handles.data() + handle_size_aligned, handle_size * 2);
364 memcpy(mapped_rchit, handles.data() + handle_size_aligned * 3, handle_size);
365}
