36 image_count = swap_chain_details.surface_capabilities.maxImageCount;

42 swap_chain_create_info.imageFormat =
43 surface_format.format; // swapchain format
44 swap_chain_create_info.imageColorSpace =
45 surface_format.colorSpace; // swapchain color space
46 swap_chain_create_info.presentMode =
47 present_mode; // swapchain presentation mode
48 swap_chain_create_info.imageExtent = extent; // swapchain image extents
49 swap_chain_create_info.minImageCount =

59 swap_chain_details.surface_capabilities

61 swap_chain_create_info.compositeAlpha =

134VkSurfaceFormatKHR VulkanSwapChain::choose_best_surface_format(
135 const std::vector<VkSurfaceFormatKHR>& formats) {
136 // best format is subjective, but I go with:
137 // Format: VK_FORMAT_R8G8B8A8_UNORM (backup-format:
138 // VK_FORMAT_B8G8R8A8_UNORM) color_space: VK_COLOR_SPACE_SRGB_NONLINEAR_KHR
139 // the condition in if means all formats are available (no restrictions)
140 if (formats.size() == 1 && formats[0].format == VK_FORMAT_UNDEFINED) {
141 return {VK_FORMAT_R8G8B8A8_UNORM, VK_COLOR_SPACE_SRGB_NONLINEAR_KHR};
142 }
143
144 // if restricted, search for optimal format
145 for (const auto& format : formats) {
146 if ((format.format == VK_FORMAT_R8G8B8A8_UNORM ||
147 format.format == VK_FORMAT_B8G8R8A8_UNORM) &&
148 format.colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR) {
149 return format;
150 }
151 }
152
153 // in case just return first one--- but really shouldn't be the case ....
154 return formats[0];
155}

157VkPresentModeKHR VulkanSwapChain::choose_best_presentation_mode(
158 const std::vector<VkPresentModeKHR>& presentation_modes) {
159 // look for mailbox presentation mode
160 for (const auto& presentation_mode : presentation_modes) {
161 if (presentation_mode == VK_PRESENT_MODE_MAILBOX_KHR) {
162 return presentation_mode;
163 }
164 }
165
166 // if can't find, use FIFO as Vulkan spec says it must be present
167 return VK_PRESENT_MODE_FIFO_KHR;
168}

170VkExtent2D VulkanSwapChain::choose_swap_extent(
171 const VkSurfaceCapabilitiesKHR& surface_capabilities) {
172 // if current extent is at numeric limits, than extent can vary. Otherwise it
173 // is size of window
174 if (surface_capabilities.currentExtent.width !=
175 std::numeric_limits<uint32_t>::max()) {
176 return surface_capabilities.currentExtent;
177
178 } else {
179 int width, height;
180 glfwGetFramebufferSize(window->get_window(), &width, &height);
181
182 // create new extent using window size
183 VkExtent2D new_extent{};
184 new_extent.width = static_cast<uint32_t>(width);
185 new_extent.height = static_cast<uint32_t>(height);
186
187 // surface also defines max and min, so make sure within boundaries bly
188 // clamping value
189 new_extent.width = std::max(
190 surface_capabilities.minImageExtent.width,
191 std::min(surface_capabilities.maxImageExtent.width, new_extent.width));
192 new_extent.height =
193 std::max(surface_capabilities.minImageExtent.height,
194 std::min(surface_capabilities.maxImageExtent.height,
195 new_extent.height));
196
197 return new_extent;
198 }
