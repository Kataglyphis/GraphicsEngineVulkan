142 size_t index_offset = 0;
143 for (size_t f = 0; f < shapes[s].mesh.num_face_vertices.size(); f++) {
144 size_t fv = size_t(shapes[s].mesh.num_face_vertices[f]);
145
146 // Loop over vertices in the face.
147 for (size_t v = 0; v < fv; v++) {
148 // access to vertex
149 tinyobj::index_t idx = shapes[s].mesh.indices[index_offset + v];
150 tinyobj::real_t vx = attrib.vertices[3 * size_t(idx.vertex_index) + 0];
151 tinyobj::real_t vy = attrib.vertices[3 * size_t(idx.vertex_index) + 1];
152 tinyobj::real_t vz = attrib.vertices[3 * size_t(idx.vertex_index) + 2];
