111void Texture::generateMipMaps(VkPhysicalDevice physical_device, VkDevice device,
112 VkCommandPool command_pool, VkQueue queue,
113 VkImage image, VkFormat image_format,
114 int32_t width, int32_t height,
115 uint32_t mip_levels) {
116 // Check if image format supports linear blitting
117 VkFormatProperties formatProperties;
118 vkGetPhysicalDeviceFormatProperties(physical_device, image_format,
119 &formatProperties);
120
121 if (!(formatProperties.optimalTilingFeatures &
122 VK_FORMAT_FEATURE_SAMPLED_IMAGE_FILTER_LINEAR_BIT)) {
123 throw std::runtime_error(
124 "Texture image format does not support linear blitting!");
125 }
126
127 VkCommandBuffer command_buffer =
128 commandBufferManager.beginCommandBuffer(device, command_pool);
129
130 VkImageMemoryBarrier barrier{};
131 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
132 barrier.image = image;
133 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
134 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
135 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
136 barrier.subresourceRange.baseArrayLayer = 0;
137 barrier.subresourceRange.layerCount = 1;
138 barrier.subresourceRange.levelCount = 1;
139
140 // TEMP VARS needed for decreasing step by step for factor 2
141 int32_t tmp_width = width;
142 int32_t tmp_height = height;
143
144 // -- WE START AT 1 !
145 for (uint32_t i = 1; i < mip_levels; i++) {
146 // WAIT for previous mip map level for being ready
147 barrier.subresourceRange.baseMipLevel = i - 1;
148 // HERE we TRANSITION for having a SRC format now
149 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
150 barrier.newLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
151 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
152 barrier.dstAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
153
154 vkCmdPipelineBarrier(command_buffer, VK_PIPELINE_STAGE_TRANSFER_BIT,
155 VK_PIPELINE_STAGE_TRANSFER_BIT, 0, 0, nullptr, 0,
156 nullptr, 1, &barrier);
157
158 // when barrier over we can now blit :)
159 VkImageBlit blit{};
160
161 // -- OFFSETS describing the 3D-dimesnion of the region
162 blit.srcOffsets[0] = {0, 0, 0};
163 blit.srcOffsets[1] = {tmp_width, tmp_height, 1};
164 blit.srcSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
165 // copy from previous level
166 blit.srcSubresource.mipLevel = i - 1;
167 blit.srcSubresource.baseArrayLayer = 0;
168 blit.srcSubresource.layerCount = 1;
169 // -- OFFSETS describing the 3D-dimesnion of the region
170 blit.dstOffsets[0] = {0, 0, 0};
171 blit.dstOffsets[1] = {tmp_width > 1 ? tmp_width / 2 : 1,
172 tmp_height > 1 ? tmp_height / 2 : 1, 1};
173 blit.dstSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
174 // -- COPY to next mipmap level
175 blit.dstSubresource.mipLevel = i;
176 blit.dstSubresource.baseArrayLayer = 0;
177 blit.dstSubresource.layerCount = 1;
178
179 vkCmdBlitImage(command_buffer, image, VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL,
180 image, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &blit,
181 VK_FILTER_LINEAR);
182
183 // REARRANGE image formats for having the correct image formats again
184 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL;
185 barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
186 barrier.srcAccessMask = VK_ACCESS_TRANSFER_READ_BIT;
187 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
188
189 vkCmdPipelineBarrier(command_buffer, VK_PIPELINE_STAGE_TRANSFER_BIT,
190 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT, 0, 0, nullptr,
191 0, nullptr, 1, &barrier);
192
193 if (tmp_width > 1) tmp_width /= 2;
194 if (tmp_height > 1) tmp_height /= 2;
195 }
196
197 barrier.subresourceRange.baseMipLevel = mip_levels - 1;
198 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
199 barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
200 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
201 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
202
203 vkCmdPipelineBarrier(command_buffer, VK_PIPELINE_STAGE_TRANSFER_BIT,
204 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT, 0, 0, nullptr, 0,
205 nullptr, 1, &barrier);
206
207 commandBufferManager.endAndSubmitCommandBuffer(device, command_pool, queue,
208 command_buffer);
