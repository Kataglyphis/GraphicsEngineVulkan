152void VulkanRenderer::drawFrame() {
153 // We need to skip one frame
154 // Due to ImGui need to call ImGui::NewFrame() again
155 // if we recreated swapchain
156 if (checkChangedFramebufferSize()) return;
157
158 /*1. Get next available image to draw to and set something to signal when
159 we're finished with the image (a semaphore) wait for given fence to signal
160 (open) from last draw before continuing*/
161 VkResult result = vkWaitForFences(device->getLogicalDevice(), 1,
162 &in_flight_fences[current_frame], VK_TRUE,
163 std::numeric_limits<uint64_t>::max());
164 ASSERT_VULKAN(result, "Failed to wait for fences!")
165 // -- GET NEXT IMAGE --
166 uint32_t image_index;
167 result = vkAcquireNextImageKHR(
168 device->getLogicalDevice(), vulkanSwapChain.getSwapChain(),
169 std::numeric_limits<uint64_t>::max(), image_available[current_frame],
170 VK_NULL_HANDLE, &image_index);
171
172 if (result == VK_ERROR_OUT_OF_DATE_KHR) {
173 // recreate_swap_chain();
174 return;
175
176 } else if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR) {
177 throw std::runtime_error("Failed to acquire next image!");
178 }
179
180 //// check if previous frame is using this image (i.e. there is its fence to
181 /// wait on)
182 if (images_in_flight_fences[image_index] != VK_NULL_HANDLE) {
183 vkWaitForFences(device->getLogicalDevice(), 1,
184 &images_in_flight_fences[image_index], VK_TRUE, UINT64_MAX);
185 }
186
187 // mark the image as now being in use by this frame
188 images_in_flight_fences[image_index] = in_flight_fences[current_frame];
189
190 VkCommandBufferBeginInfo buffer_begin_info{};
191 buffer_begin_info.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
192 buffer_begin_info.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
193 // start recording commands to command buffer
194 result =
195 vkBeginCommandBuffer(command_buffers[image_index], &buffer_begin_info);
196 ASSERT_VULKAN(result, "Failed to start recording a command buffer!")
197
198 update_uniform_buffers(image_index);
199
200 GUIRendererSharedVars& guiRendererSharedVars =
201 gui->getGuiRendererSharedVars();
202 if (guiRendererSharedVars.raytracing)
203 update_raytracing_descriptor_set(image_index);
204
205 record_commands(image_index);
206
207 // stop recording to command buffer
208 result = vkEndCommandBuffer(command_buffers[image_index]);
209 ASSERT_VULKAN(result, "Failed to stop recording a command buffer!")
210
211 // 2. Submit command buffer to queue for execution, making sure it waits for
212 // the image to be signalled as available before drawing and signals when it
213 // has finished rendering
214 // -- SUBMIT COMMAND BUFFER TO RENDER --
215 VkSubmitInfo submit_info{};
216 submit_info.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
217 submit_info.waitSemaphoreCount = 1; // number of semaphores to wait on
218 submit_info.pWaitSemaphores =
219 &image_available[current_frame]; // list of semaphores to wait on
220
221 VkPipelineStageFlags wait_stages = {
222
223 VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT /*|
224 VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT |
225 VK_PIPELINE_STAGE_2_RAY_TRACING_SHADER_BIT_KHR*/
226
227 };
228
229 submit_info.pWaitDstStageMask =
230 &wait_stages; // stages to check semaphores at
231
232 submit_info.commandBufferCount = 1; // number of command buffers to submit
233 submit_info.pCommandBuffers =
234 &command_buffers[image_index]; // command buffer to submit
235 submit_info.signalSemaphoreCount = 1; // number of semaphores to signal
236 submit_info.pSignalSemaphores =
237 &render_finished[current_frame]; // semaphores to signal when command
238 // buffer finishes
239
240 result = vkResetFences(device->getLogicalDevice(), 1,
241 &in_flight_fences[current_frame]);
242 ASSERT_VULKAN(result, "Failed to reset fences!")
243
244 // submit command buffer to queue
245 result = vkQueueSubmit(device->getGraphicsQueue(), 1, &submit_info,
246 in_flight_fences[current_frame]);
247 ASSERT_VULKAN(result, "Failed to submit command buffer to queue!")
248
249 // 3. Present image to screen when it has signalled finished rendering
250 // -- PRESENT RENDERED IMAGE TO SCREEN --
251 VkPresentInfoKHR present_info{};
252 present_info.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
253 present_info.waitSemaphoreCount = 1; // number of semaphores to wait on
254 present_info.pWaitSemaphores =
255 &render_finished[current_frame]; // semaphores to wait on
256 present_info.swapchainCount = 1; // number of swapchains to present to
257 const VkSwapchainKHR swapchain = vulkanSwapChain.getSwapChain();
258 present_info.pSwapchains = &swapchain; // swapchains to present images to
259 present_info.pImageIndices =
260 &image_index; // index of images in swapchain to present
261
262 result = vkQueuePresentKHR(device->getPresentationQueue(), &present_info);
263
264 if (result == VK_ERROR_OUT_OF_DATE_KHR) {
265 // recreate_swap_chain();
266 return;
267
268 } else if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR) {
269 throw std::runtime_error("Failed to acquire next image!");
270 }
271
272 if (result != VK_SUCCESS) {
273 throw std::runtime_error("Failed to submit to present queue!");
274 }
275
276 current_frame = (current_frame + 1) % MAX_FRAME_DRAWS;
277}
