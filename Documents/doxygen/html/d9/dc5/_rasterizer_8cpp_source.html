44void Rasterizer::recordCommands(
45 VkCommandBuffer& commandBuffer, uint32_t image_index, Scene* scene,
46 const std::vector<VkDescriptorSet>& descriptorSets) {
47 // information about how to begin a render pass (only needed for graphical
48 // applications)
49 VkRenderPassBeginInfo render_pass_begin_info{};
50 render_pass_begin_info.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
51 render_pass_begin_info.renderPass = render_pass;
52 render_pass_begin_info.renderArea.offset = {0, 0};
53 const VkExtent2D& swap_chain_extent = vulkanSwapChain->getSwapChainExtent();
54 render_pass_begin_info.renderArea.extent = swap_chain_extent;
55
56 // make sure the order you put the values into the array matches with the
57 // attchment order you have defined previous
58 std::array<VkClearValue, 2> clear_values = {};
59 clear_values[0].color = {0.2f, 0.65f, 0.4f, 1.0f};
60 clear_values[1].depthStencil = {1.0f, 0};
61
62 render_pass_begin_info.pClearValues = clear_values.data();
63 render_pass_begin_info.clearValueCount =
64 static_cast<uint32_t>(clear_values.size());
65 render_pass_begin_info.framebuffer = framebuffer[image_index];
66
67 // begin render pass
68 vkCmdBeginRenderPass(commandBuffer, &render_pass_begin_info,
69 VK_SUBPASS_CONTENTS_INLINE);
70
71 // bind pipeline to be used in render pass
72 vkCmdBindPipeline(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS,
73 graphics_pipeline);
74
75 for (uint32_t m = 0; m < static_cast<uint32_t>(scene->getModelCount()); m++) {
76 // for GCC doen't allow references on rvalues go like that ...
77 pushConstant.model = scene->getModelMatrix(0);
78 // just "Push" constants to given shader stage directly (no buffer)
79 vkCmdPushConstants(
80 commandBuffer, pipeline_layout,
81 VK_SHADER_STAGE_VERTEX_BIT, // stage to push constants to
82 0, // offset to push constants to update
83 sizeof(PushConstantRasterizer), // size of data being pushed
84 &pushConstant); // using model of current mesh (can be array)
85
86 for (unsigned int k = 0; k < scene->getMeshCount(m); k++) {
87 // list of vertex buffers we want to draw
88 VkBuffer vertex_buffers[] = {
89 scene->getVertexBuffer(m, k)}; // buffers to bind
90 VkDeviceSize offsets[] = {0};
91 vkCmdBindVertexBuffers(
92 commandBuffer, 0, 1, vertex_buffers,
93 offsets); // command to bind vertex buffer before drawing with them
94
95 // bind mesh index buffer with 0 offset and using the uint32 type
96 vkCmdBindIndexBuffer(commandBuffer, scene->getIndexBuffer(m, k), 0,
97 VK_INDEX_TYPE_UINT32);
98
99 // bind descriptor sets
100 vkCmdBindDescriptorSets(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS,
101 pipeline_layout, 0,
102 static_cast<uint32_t>(descriptorSets.size()),
103 descriptorSets.data(), 0, nullptr);
104
105 // execute pipeline
106 vkCmdDrawIndexed(commandBuffer,
107 static_cast<uint32_t>(scene->getIndexCount(m, k)), 1, 0,
108 0, 0);
109 }
110 }
111
112 // end render pass
113 vkCmdEndRenderPass(commandBuffer);
114}

134void Rasterizer::createRenderPass() {
135 // Color attachment of render pass
136 VkAttachmentDescription color_attachment{};
137 const VkFormat& swap_chain_image_format =
138 vulkanSwapChain->getSwapChainFormat();
139 color_attachment.format =
140 swap_chain_image_format; // format to use for attachment
141 color_attachment.samples =
142 VK_SAMPLE_COUNT_1_BIT; // number of samples to write for multisampling
143 color_attachment.loadOp =
144 VK_ATTACHMENT_LOAD_OP_CLEAR; // describes what to do with attachment
145 // before rendering
146 color_attachment.storeOp =
147 VK_ATTACHMENT_STORE_OP_STORE; // describes what to do with attachment
148 // after rendering
149 color_attachment.stencilLoadOp =
150 VK_ATTACHMENT_LOAD_OP_DONT_CARE; // describes what to do with stencil
151 // before rendering
152 color_attachment.stencilStoreOp =
153 VK_ATTACHMENT_STORE_OP_DONT_CARE; // describes what to do with stencil
154 // after rendering
155
156 // framebuffer data will be stored as an image, but images can be given
157 // different layouts to give optimal use for certain operations
158 color_attachment.initialLayout =
159 VK_IMAGE_LAYOUT_GENERAL; // image data layout before render pass starts
160 color_attachment.finalLayout =
161 VK_IMAGE_LAYOUT_GENERAL; // image data layout after render pass (to
162 // change to)
163
164 // depth attachment of render pass
165 VkAttachmentDescription depth_attachment{};
166 depth_attachment.format = choose_supported_format(
167 device->getPhysicalDevice(),
168 {VK_FORMAT_D32_SFLOAT_S8_UINT, VK_FORMAT_D32_SFLOAT,
169 VK_FORMAT_D24_UNORM_S8_UINT},
170 VK_IMAGE_TILING_OPTIMAL, VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT);
171
172 depth_attachment.samples = VK_SAMPLE_COUNT_1_BIT;
173 depth_attachment.loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
174 depth_attachment.storeOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
175 depth_attachment.stencilLoadOp = VK_ATTACHMENT_LOAD_OP_DONT_CARE;
176 depth_attachment.stencilStoreOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
177 depth_attachment.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
178 depth_attachment.finalLayout =
179 VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
180
181 // attachment reference uses an attachment index that refers to index in the
182 // attachment list passed to renderPassCreateInfo
183 VkAttachmentReference color_attachment_reference{};
184 color_attachment_reference.attachment = 0;
185 color_attachment_reference.layout = VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL;
186
187 // attachment reference
188 VkAttachmentReference depth_attachment_reference{};
189 depth_attachment_reference.attachment = 1;
190 depth_attachment_reference.layout =
191 VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
192
193 // information about a particular subpass the render pass is using
194 VkSubpassDescription subpass{};
195 subpass.pipelineBindPoint =
196 VK_PIPELINE_BIND_POINT_GRAPHICS; // pipeline type subpass is to be bound
197 // to
198 subpass.colorAttachmentCount = 1;
199 subpass.pColorAttachments = &color_attachment_reference;
200 subpass.pDepthStencilAttachment = &depth_attachment_reference;
201
202 // need to determine when layout transitions occur using subpass dependencies
203 std::array<VkSubpassDependency, 1> subpass_dependencies;
204
205 // conversion from VK_IMAGE_LAYOUT_UNDEFINED to
206 // VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL transition must happen after ....
207 subpass_dependencies[0].srcSubpass =
208 VK_SUBPASS_EXTERNAL; // subpass index (VK_SUBPASS_EXTERNAL = Special
209 // value meaning outside of renderpass)
210 subpass_dependencies[0].srcStageMask =
211 VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT; // pipeline stage
212 subpass_dependencies[0].srcAccessMask =
213 0; // stage access mask (memory access)
214
215 // but must happen before ...
216 subpass_dependencies[0].dstSubpass = 0;
217 subpass_dependencies[0].dstStageMask =
218 VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
219 subpass_dependencies[0].dstAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
220 subpass_dependencies[0].dependencyFlags = 0; // VK_DEPENDENCY_BY_REGION_BIT;
221
222 std::array<VkAttachmentDescription, 2> render_pass_attachments = {
223 color_attachment, depth_attachment};
224
225 // create info for render pass
226 VkRenderPassCreateInfo render_pass_create_info{};
227 render_pass_create_info.sType = VK_STRUCTURE_TYPE_RENDER_PASS_CREATE_INFO;
228 render_pass_create_info.attachmentCount =
229 static_cast<uint32_t>(render_pass_attachments.size());
230 render_pass_create_info.pAttachments = render_pass_attachments.data();
231 render_pass_create_info.subpassCount = 1;
232 render_pass_create_info.pSubpasses = &subpass;
233 render_pass_create_info.dependencyCount =
234 static_cast<uint32_t>(subpass_dependencies.size());
235 render_pass_create_info.pDependencies = subpass_dependencies.data();
236
237 VkResult result =
238 vkCreateRenderPass(device->getLogicalDevice(), &render_pass_create_info,
239 nullptr, &render_pass);
240 ASSERT_VULKAN(result, "Failed to create render pass!")
241}

275void Rasterizer::createTextures(VkCommandPool& commandPool) {
276 offscreenTextures.resize(vulkanSwapChain->getNumberSwapChainImages());
277
278 VkCommandBuffer cmdBuffer = commandBufferManager.beginCommandBuffer(
279 device->getLogicalDevice(), commandPool);
280
281 for (uint32_t index = 0;
282 index <
283 static_cast<uint32_t>(vulkanSwapChain->getNumberSwapChainImages());
284 index++) {
285 Texture texture{};
286 const VkExtent2D& swap_chain_extent = vulkanSwapChain->getSwapChainExtent();
287 const VkFormat& swap_chain_image_format =
288 vulkanSwapChain->getSwapChainFormat();
289
290 texture.createImage(
291 device, swap_chain_extent.width, swap_chain_extent.height, 1,
292 swap_chain_image_format, VK_IMAGE_TILING_OPTIMAL,
293 VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_SAMPLED_BIT |
294 VK_IMAGE_USAGE_STORAGE_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT,
295 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
296
297 texture.createImageView(device, swap_chain_image_format,
298 VK_IMAGE_ASPECT_COLOR_BIT, 1);
299
300 // --- WE NEED A DIFFERENT LAYOUT FOR USAGE
301 VulkanImage& image = texture.getVulkanImage();
302 image.transitionImageLayout(cmdBuffer, VK_IMAGE_LAYOUT_UNDEFINED,
303 VK_IMAGE_LAYOUT_GENERAL, 1,
304 VK_IMAGE_ASPECT_COLOR_BIT);
305
306 offscreenTextures[index] = texture;
307 }
308
309 VkFormat depth_format = choose_supported_format(
310 device->getPhysicalDevice(),
311 {VK_FORMAT_D32_SFLOAT_S8_UINT, VK_FORMAT_D32_SFLOAT,
312 VK_FORMAT_D24_UNORM_S8_UINT},
313 VK_IMAGE_TILING_OPTIMAL, VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT);
314
315 // create depth buffer image
316 // MIP LEVELS: for depth texture we only want 1 level :)
317 const VkExtent2D& swap_chain_extent = vulkanSwapChain->getSwapChainExtent();
318 depthBufferImage.createImage(device, swap_chain_extent.width,
319 swap_chain_extent.height, 1, depth_format,
320 VK_IMAGE_TILING_OPTIMAL,
321 VK_IMAGE_USAGE_DEPTH_STENCIL_ATTACHMENT_BIT,
322 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
323
324 // depth buffer image view
325 // MIP LEVELS: for depth texture we only want 1 level :)
326 depthBufferImage.createImageView(
327 device, depth_format,
328 VK_IMAGE_ASPECT_DEPTH_BIT | VK_IMAGE_ASPECT_STENCIL_BIT, 1);
329
330 // --- WE NEED A DIFFERENT LAYOUT FOR USAGE
331 VulkanImage& vulkanImage = depthBufferImage.getVulkanImage();
332 vulkanImage.transitionImageLayout(
333 device->getLogicalDevice(), device->getGraphicsQueue(), commandPool,
334 VK_IMAGE_LAYOUT_UNDEFINED,
335 VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL,
336 VK_IMAGE_ASPECT_DEPTH_BIT | VK_IMAGE_ASPECT_STENCIL_BIT, 1);
337
338 commandBufferManager.endAndSubmitCommandBuffer(
339 device->getLogicalDevice(), commandPool, device->getGraphicsQueue(),
340 cmdBuffer);
341}

343void Rasterizer::createGraphicsPipeline(
344 const std::vector<VkDescriptorSetLayout>& descriptorSetLayouts) {
345 std::stringstream rasterizer_shader_dir;
346 std::filesystem::path cwd = std::filesystem::current_path();
347 rasterizer_shader_dir << cwd.string();
348 rasterizer_shader_dir << RELATIVE_RESOURCE_PATH;
349 rasterizer_shader_dir << "Shaders/rasterizer/";
350
351 ShaderHelper shaderHelper;
352 shaderHelper.compileShader(rasterizer_shader_dir.str(), "shader.vert");
353 shaderHelper.compileShader(rasterizer_shader_dir.str(), "shader.frag");
354
355 File vertexFile(
356 shaderHelper.getShaderSpvDir(rasterizer_shader_dir.str(), "shader.vert"));
357 File fragmentFile(
358 shaderHelper.getShaderSpvDir(rasterizer_shader_dir.str(), "shader.frag"));
359 std::vector<char> vertex_shader_code = vertexFile.readCharSequence();
360 std::vector<char> fragment_shader_code = fragmentFile.readCharSequence();
361
362 // build shader modules to link to graphics pipeline
363 VkShaderModule vertex_shader_module =
364 shaderHelper.createShaderModule(device, vertex_shader_code);
365 VkShaderModule fragment_shader_module =
366 shaderHelper.createShaderModule(device, fragment_shader_code);
367
368 // shader stage creation information
369 // vertex stage creation information
370 VkPipelineShaderStageCreateInfo vertex_shader_create_info{};
371 vertex_shader_create_info.sType =
372 VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
373 vertex_shader_create_info.stage = VK_SHADER_STAGE_VERTEX_BIT;
374 vertex_shader_create_info.module = vertex_shader_module;
375 vertex_shader_create_info.pName = "main";
376
377 // fragment stage creation information
378 VkPipelineShaderStageCreateInfo fragment_shader_create_info{};
379 fragment_shader_create_info.sType =
380 VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
381 fragment_shader_create_info.stage = VK_SHADER_STAGE_FRAGMENT_BIT;
382 fragment_shader_create_info.module = fragment_shader_module;
383 fragment_shader_create_info.pName = "main";
384
385 std::vector<VkPipelineShaderStageCreateInfo> shader_stages = {
386 vertex_shader_create_info, fragment_shader_create_info};
387
388 // how the data for a single vertex (including info such as position, color,
389 // texture coords, normals, etc) is as a whole
390 VkVertexInputBindingDescription binding_description{};
391 binding_description.binding = 0;
392 binding_description.stride = sizeof(Vertex);
393 binding_description.inputRate =
394 VK_VERTEX_INPUT_RATE_VERTEX; // how to move between data after each
395 // vertex.
396
397 // how the data for an attribute is defined within a vertex
398 std::array<VkVertexInputAttributeDescription, 4> attribute_describtions =
399 vertex::getVertexInputAttributeDesc();
400
401 // CREATE PIPELINE
402 // 1.) Vertex input
403 VkPipelineVertexInputStateCreateInfo vertex_input_create_info{};
404 vertex_input_create_info.sType =
405 VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
406 vertex_input_create_info.vertexBindingDescriptionCount = 1;
407 vertex_input_create_info.pVertexBindingDescriptions = &binding_description;
408 vertex_input_create_info.vertexAttributeDescriptionCount =
409 static_cast<uint32_t>(attribute_describtions.size());
410 vertex_input_create_info.pVertexAttributeDescriptions =
411 attribute_describtions.data();
412
413 // input assembly
414 VkPipelineInputAssemblyStateCreateInfo input_assembly{};
415 input_assembly.sType =
416 VK_STRUCTURE_TYPE_PIPELINE_INPUT_ASSEMBLY_STATE_CREATE_INFO;
417 input_assembly.topology = VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST;
418 input_assembly.primitiveRestartEnable = VK_FALSE;
419
420 // viewport & scissor
421 // create a viewport info struct
422 VkViewport viewport{};
423 viewport.x = 0.0f;
424 viewport.y = 0.0f;
425 const VkExtent2D& swap_chain_extent = vulkanSwapChain->getSwapChainExtent();
426 viewport.width = (float)swap_chain_extent.width;
427 viewport.height = (float)swap_chain_extent.height;
428 viewport.minDepth = 0.0f;
429 viewport.maxDepth = 1.0f;
430
431 // create a scissor info struct
432 VkRect2D scissor{};
433 scissor.offset = {0, 0};
434 scissor.extent = swap_chain_extent;
435
436 VkPipelineViewportStateCreateInfo viewport_state_create_info{};
437 viewport_state_create_info.sType =
438 VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
439 viewport_state_create_info.viewportCount = 1;
440 viewport_state_create_info.pViewports = &viewport;
441 viewport_state_create_info.scissorCount = 1;
442 viewport_state_create_info.pScissors = &scissor;
443
444 // RASTERIZER
445 VkPipelineRasterizationStateCreateInfo rasterizer_create_info{};
446 rasterizer_create_info.sType =
447 VK_STRUCTURE_TYPE_PIPELINE_RASTERIZATION_STATE_CREATE_INFO;
448 rasterizer_create_info.depthClampEnable = VK_FALSE;
449 rasterizer_create_info.rasterizerDiscardEnable = VK_FALSE;
450 rasterizer_create_info.polygonMode = VK_POLYGON_MODE_FILL;
451 rasterizer_create_info.lineWidth = 1.0f;
452 rasterizer_create_info.cullMode = VK_CULL_MODE_BACK_BIT; //
453 // winding to determine which side is front; y-coordinate is inverted in
454 // comparison to OpenGL
455 rasterizer_create_info.frontFace = VK_FRONT_FACE_COUNTER_CLOCKWISE;
456 rasterizer_create_info.depthBiasClamp = VK_FALSE;
457
458 // -- MULTISAMPLING --
459 VkPipelineMultisampleStateCreateInfo multisample_create_info{};
460 multisample_create_info.sType =
461 VK_STRUCTURE_TYPE_PIPELINE_MULTISAMPLE_STATE_CREATE_INFO;
462 multisample_create_info.sampleShadingEnable = VK_FALSE;
463 multisample_create_info.rasterizationSamples = VK_SAMPLE_COUNT_1_BIT;
464
465 // -- BLENDING --
466 // blend attachment state
467 VkPipelineColorBlendAttachmentState color_state{};
468 color_state.colorWriteMask =
469 VK_COLOR_COMPONENT_R_BIT | VK_COLOR_COMPONENT_G_BIT |
470 VK_COLOR_COMPONENT_B_BIT | VK_COLOR_COMPONENT_A_BIT;
471
472 color_state.blendEnable = VK_TRUE;
473 // blending uses equation: (srcColorBlendFactor * new_color) color_blend_op
474 // (dstColorBlendFactor * old_color)
475 color_state.srcColorBlendFactor = VK_BLEND_FACTOR_SRC_ALPHA;
476 color_state.dstColorBlendFactor = VK_BLEND_FACTOR_ONE_MINUS_SRC_ALPHA;
477 color_state.colorBlendOp = VK_BLEND_OP_ADD;
478 color_state.srcAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
479 color_state.dstAlphaBlendFactor = VK_BLEND_FACTOR_ZERO;
480 color_state.alphaBlendOp = VK_BLEND_OP_ADD;
481
482 VkPipelineColorBlendStateCreateInfo color_blending_create_info{};
483 color_blending_create_info.sType =
484 VK_STRUCTURE_TYPE_PIPELINE_COLOR_BLEND_STATE_CREATE_INFO;
485 color_blending_create_info.logicOpEnable = VK_FALSE;
486 color_blending_create_info.attachmentCount = 1;
487 color_blending_create_info.pAttachments = &color_state;
488
489 // -- PIPELINE LAYOUT --
490 VkPipelineLayoutCreateInfo pipeline_layout_create_info{};
491 pipeline_layout_create_info.sType =
492 VK_STRUCTURE_TYPE_PIPELINE_LAYOUT_CREATE_INFO;
493 pipeline_layout_create_info.setLayoutCount =
494 static_cast<uint32_t>(descriptorSetLayouts.size());
495 pipeline_layout_create_info.pSetLayouts = descriptorSetLayouts.data();
496 pipeline_layout_create_info.pushConstantRangeCount = 1;
497 pipeline_layout_create_info.pPushConstantRanges = &push_constant_range;
498
499 // create pipeline layout
500 VkResult result = vkCreatePipelineLayout(device->getLogicalDevice(),
501 &pipeline_layout_create_info,
502 nullptr, &pipeline_layout);
503 ASSERT_VULKAN(result, "Failed to create pipeline layout!")
504
505 // -- DEPTH STENCIL TESTING --
506 VkPipelineDepthStencilStateCreateInfo depth_stencil_create_info{};
507 depth_stencil_create_info.sType =
508 VK_STRUCTURE_TYPE_PIPELINE_DEPTH_STENCIL_STATE_CREATE_INFO;
509 depth_stencil_create_info.depthTestEnable = VK_TRUE;
510 depth_stencil_create_info.depthWriteEnable = VK_TRUE;
511 depth_stencil_create_info.depthCompareOp = VK_COMPARE_OP_LESS;
512 depth_stencil_create_info.depthBoundsTestEnable = VK_FALSE;
513 depth_stencil_create_info.stencilTestEnable = VK_FALSE;
514
515 // -- GRAPHICS PIPELINE CREATION --
516 VkGraphicsPipelineCreateInfo graphics_pipeline_create_info{};
517 graphics_pipeline_create_info.sType =
518 VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
519 graphics_pipeline_create_info.stageCount =
520 static_cast<uint32_t>(shader_stages.size());
521 graphics_pipeline_create_info.pStages = shader_stages.data();
522 graphics_pipeline_create_info.pVertexInputState = &vertex_input_create_info;
523 graphics_pipeline_create_info.pInputAssemblyState = &input_assembly;
524 graphics_pipeline_create_info.pViewportState = &viewport_state_create_info;
525 graphics_pipeline_create_info.pDynamicState = nullptr;
526 graphics_pipeline_create_info.pRasterizationState = &rasterizer_create_info;
527 graphics_pipeline_create_info.pMultisampleState = &multisample_create_info;
528 graphics_pipeline_create_info.pColorBlendState = &color_blending_create_info;
529 graphics_pipeline_create_info.pDepthStencilState = &depth_stencil_create_info;
530 graphics_pipeline_create_info.layout = pipeline_layout;
531 graphics_pipeline_create_info.renderPass = render_pass;
532 graphics_pipeline_create_info.subpass = 0;
533
534 // pipeline derivatives : can create multiple pipelines that derive from one
535 // another for optimization
536 graphics_pipeline_create_info.basePipelineHandle = VK_NULL_HANDLE;
537 graphics_pipeline_create_info.basePipelineIndex = -1;
