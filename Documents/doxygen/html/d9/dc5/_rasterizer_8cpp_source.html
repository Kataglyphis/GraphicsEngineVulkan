54  render_pass_begin_info.renderArea.extent = swap_chain_extent;
55

57  // attchment order you have defined previous
58  std::array<VkClearValue, 2> clear_values = {};
59  clear_values[0].color = {0.2f, 0.65f, 0.4f, 1.0f};
60  clear_values[1].depthStencil = {1.0f, 0};
61

63  render_pass_begin_info.clearValueCount =
64  static_cast<uint32_t>(clear_values.size());
65  render_pass_begin_info.framebuffer = framebuffer[image_index];
66

70

73  graphics_pipeline);
74 
75  for (uint32_t m = 0; m < static_cast<uint32_t>(scene->getModelCount()); m++) {
76  // for GCC doen't allow references on rvalues go like that ...
77  pushConstant.model = scene->getModelMatrix(0);
78  // just "Push" constants to given shader stage directly (no buffer)
79  vkCmdPushConstants(
80  commandBuffer, pipeline_layout,
81  VK_SHADER_STAGE_VERTEX_BIT, // stage to push constants to
82  0, // offset to push constants to update

84  &pushConstant); // using model of current mesh (can be array)
85 
86  for (unsigned int k = 0; k < scene->getMeshCount(m); k++) {

88  VkBuffer vertex_buffers[] = {

90  VkDeviceSize offsets[] = {0};

92  commandBuffer, 0, 1, vertex_buffers,
93  offsets); // command to bind vertex buffer before drawing with them
94

96  vkCmdBindIndexBuffer(commandBuffer, scene->getIndexBuffer(m, k), 0,
97  VK_INDEX_TYPE_UINT32);
98

100  vkCmdBindDescriptorSets(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS,

104

106  vkCmdDrawIndexed(commandBuffer,
107  static_cast<uint32_t>(scene->getIndexCount(m, k)), 1, 0,

148  // after rendering

151  // before rendering

154  // after rendering

159  VK_IMAGE_LAYOUT_GENERAL; // image data layout before render pass starts
160  color_attachment.finalLayout =
161  VK_IMAGE_LAYOUT_GENERAL; // image data layout after render pass (to
162  // change to)

166  depth_attachment.format = choose_supported_format(

171 
172  depth_attachment.samples = VK_SAMPLE_COUNT_1_BIT;
173  depth_attachment.loadOp = VK_ATTACHMENT_LOAD_OP_CLEAR;
174  depth_attachment.storeOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;
175  depth_attachment.stencilLoadOp = VK_ATTACHMENT_LOAD_OP_DONT_CARE;
176  depth_attachment.stencilStoreOp = VK_ATTACHMENT_STORE_OP_DONT_CARE;

178  depth_attachment.finalLayout =
179  VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;

184  color_attachment_reference.attachment = 0;
185  color_attachment_reference.layout = VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL;
186

188  VkAttachmentReference depth_attachment_reference{};
189  depth_attachment_reference.attachment = 1;
190  depth_attachment_reference.layout =
191  VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL;
192 
193  // information about a particular subpass the render pass is using
194  VkSubpassDescription subpass{};
195  subpass.pipelineBindPoint =
196  VK_PIPELINE_BIND_POINT_GRAPHICS; // pipeline type subpass is to be bound
197  // to
198  subpass.colorAttachmentCount = 1;
199  subpass.pColorAttachments = &color_attachment_reference;
200  subpass.pDepthStencilAttachment = &depth_attachment_reference;
201 
202  // need to determine when layout transitions occur using subpass dependencies
203  std::array<VkSubpassDependency, 1> subpass_dependencies;
204 
205  // conversion from VK_IMAGE_LAYOUT_UNDEFINED to
206  // VK_IMAGE_LAYOUT_COLOR_ATTACHMENT_OPTIMAL transition must happen after ....
207  subpass_dependencies[0].srcSubpass =
208  VK_SUBPASS_EXTERNAL; // subpass index (VK_SUBPASS_EXTERNAL = Special
209  // value meaning outside of renderpass)
210  subpass_dependencies[0].srcStageMask =
211  VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT; // pipeline stage
212  subpass_dependencies[0].srcAccessMask =
213  0; // stage access mask (memory access)
214 
215  // but must happen before ...
216  subpass_dependencies[0].dstSubpass = 0;
217  subpass_dependencies[0].dstStageMask =
218  VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
219  subpass_dependencies[0].dstAccessMask = VK_ACCESS_COLOR_ATTACHMENT_WRITE_BIT;
220  subpass_dependencies[0].dependencyFlags = 0; // VK_DEPENDENCY_BY_REGION_BIT;
221 
222  std::array<VkAttachmentDescription, 2> render_pass_attachments = {
223  color_attachment, depth_attachment};
224 
225  // create info for render pass
226  VkRenderPassCreateInfo render_pass_create_info{};
227  render_pass_create_info.sType = VK_STRUCTURE_TYPE_RENDER_PASS_CREATE_INFO;
228  render_pass_create_info.attachmentCount =
229  static_cast<uint32_t>(render_pass_attachments.size());
230  render_pass_create_info.pAttachments = render_pass_attachments.data();
231  render_pass_create_info.subpassCount = 1;
232  render_pass_create_info.pSubpasses = &subpass;
233  render_pass_create_info.dependencyCount =
234  static_cast<uint32_t>(subpass_dependencies.size());
235  render_pass_create_info.pDependencies = subpass_dependencies.data();
236 
237  VkResult result =
238  vkCreateRenderPass(device->getLogicalDevice(), &render_pass_create_info,
239  nullptr, &render_pass);
240  ASSERT_VULKAN(result, "Failed to create render pass!")
241 }
